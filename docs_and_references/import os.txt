LEGAL_NOTE = """
LƯU Ý QUAN TRỌNG CHO LLM:
- Đây là văn bản pháp luật. Không được diễn giải lại khi chunk.
- Mỗi chunk phải giữ nguyên nội dung gốc 100%, không tự ý rút gọn.
- Mỗi 'Điều' phải được giữ nguyên phần tiêu đề + nội dung ngay sau nó.
- Nếu chunk quá dài và cần chia nhỏ → mỗi phần phải giữ lại tiêu đề Điều ở đầu.
- Tuyệt đối không tách giữa các khoản theo kiểu làm mất tính pháp lý.
"""


# ============================================================
# CUSTOM LEGAL CHUNKING + LOG
# ============================================================
def custom_legal_chunking(
    tokenizer: Tokenizer,
    content: str,
    chunk_token_size: int = 4000,
    chunk_overlap_token_size: int = 100,
    split_by_character: str | None = None,
    split_by_character_only: bool = False,
) -> List[Dict[str, Any]]:

    print("\n=== BẮT ĐẦU CHUNKING TÀI LIỆU PHÁP LUẬT ===")
    print("NOTE:", LEGAL_NOTE)

    # Regex mạnh nhận diện tất cả dạng “Điều x”
    pattern = r"(?=Điều\s+\d+[\.: -])"

    print("[LOG] Tách văn bản theo regex:", pattern)

    raw_chunks = re.split(pattern, content)
    results = []
    chunk_index = 0

    for raw in raw_chunks:
        chunk = raw.strip()
        if not chunk:
            continue

        # Skip NOTE trống
        if len(chunk) < 3:
            continue

        print(f"\n--- CHUNK #{chunk_index} (Điều) ---")
        print(chunk[:120].replace("\n", " ") + " ...")

        tokens = tokenizer.encode(chunk)
        token_count = len(tokens)

        print(f"[LOG] Token count: {token_count}")

        # -------------------------------------------------
        # CASE 1: Chunk nhỏ → dùng nguyên
        # -------------------------------------------------
        if token_count <= chunk_token_size:
            print("[LOG] ✔ Chunk đạt chuẩn, thêm vào danh sách.")
            results.append({
                "tokens": token_count,
                "content": chunk,
                "chunk_order_index": chunk_index,
                "note": LEGAL_NOTE
            })
            chunk_index += 1
            continue

        # -------------------------------------------------
        # CASE 2: Chunk quá lớn → SUB-CHUNK
        # -------------------------------------------------
        print("[WARNING] Điều quá dài, tiến hành sub-chunking!")

        lines = chunk.split("\n")
        header = lines[0]  # "Điều X..."
        body = "\n".join(lines[1:]).strip()

        temp = ""
        current_tokens = []

        for para in body.split("\n"):
            test = (temp + "\n" + para).strip()
            t = tokenizer.encode(test)

            if len(t) > chunk_token_size:
                # Save chunk cũ
                print("[LOG] → Xuất sub-chunk")
                results.append({
                    "tokens": len(current_tokens),
                    "content": f"{header}\n{temp.strip()}",
                    "chunk_order_index": chunk_index,
                    "note": LEGAL_NOTE
                })
                chunk_index += 1

                temp = para
                current_tokens = tokenizer.encode(para)
            else:
                temp = test
                current_tokens = t

        # Save sub-chunk cuối
        print("[LOG] → Lưu sub-chunk cuối")
        results.append({
            "tokens": len(current_tokens),
            "content": f"{header}\n{temp.strip()}",
            "chunk_order_index": chunk_index,
            "note": LEGAL_NOTE
        })
        chunk_index += 1

    print("\n=== HOÀN TẤT CHUNKING ===")
    print(f"Tổng số chunk: {len(results)}")

    return results
